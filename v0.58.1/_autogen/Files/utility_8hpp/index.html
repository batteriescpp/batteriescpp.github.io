#define BATT_FORWARD(x) std::forward<decltype(x)>(x)

template <typename T>
T make_copy(const T& value)
{
    return value;
}

// BATT_SINK(val) - turns into BATT_FORWARD(val) if val is an rvalue-expression; otherwise, turns into
// batt::make_copy(val).
//
template <typename T, typename = std::enable_if_t<std::is_same_v<T, std::decay_t<T>>>>
T&& sink(T&& value)
{
    return std::forward<T>(value);
}

template <typename T>
T sink(const T& value)
{
    return make_copy(value);
}

#define BATT_SINK(expr) ::batt::sink(BATT_FORWARD(expr))

template <typename T, typename = std::enable_if_t<!std::is_same_v<T, void>>>
T make_default()
{
    return T{};
}

template <typename T, typename = std::enable_if_t<std::is_same_v<T, void>>, typename = void>
void make_default()
{
}

struct DefaultInitialized {
    template <typename T>
    /*implicit*/ operator T() const noexcept(noexcept(make_default<T>()))
    {
        return make_default<T>();
    }
};

template <typename T>
decltype(auto) unwrap_ref(T&& obj)
{
    return BATT_FORWARD(obj);
}

#define BATT_UNWRAP_REF_STD_REFERENCE_WRAPPER(qualifiers)                                                    \
    template <typename T>                                                                                    \
    decltype(auto) unwrap_ref(std::reference_wrapper<T> qualifiers wrapper)                                  \
    {                                                                                                        \
        return BATT_FORWARD(wrapper).get();                                                                  \
    }

namespace detail {

template <typename Fn, typename ArgsTuple, usize... kIndex>
decltype(auto) rotate_args_right_impl(Fn&& fn, ArgsTuple&& args_tuple, std::index_sequence<kIndex...>)
{
    constexpr usize kLastIndex = sizeof...(kIndex);

    return BATT_FORWARD(fn)(std::get<kLastIndex>(BATT_FORWARD(args_tuple)),
                            std::get<kIndex>(BATT_FORWARD(args_tuple))...);
}

}  // namespace detail

template <typename Fn, typename... Args>
decltype(auto) rotate_args_right(Fn&& fn, Args&&... args)
{
    return detail::rotate_args_right_impl(BATT_FORWARD(fn), std::forward_as_tuple(BATT_FORWARD(args)...),
                                          std::make_index_sequence<sizeof...(args) - 1>{});
}

namespace detail {

template <typename Fn, typename ArgsTuple, usize... kIndex>
decltype(auto) rotate_args_left_impl(Fn&& fn, ArgsTuple&& args_tuple, std::index_sequence<kIndex...>)
{
    return BATT_FORWARD(fn)(std::get<kIndex + 1>(BATT_FORWARD(args_tuple))...,
                            std::get<0>(BATT_FORWARD(args_tuple)));
}

}  // namespace detail

template <typename Fn, typename... Args>
decltype(auto) rotate_args_left(Fn&& fn, Args&&... args)
{
    return detail::rotate_args_left_impl(BATT_FORWARD(fn), std::forward_as_tuple(BATT_FORWARD(args)...),
                                         std::make_index_sequence<sizeof...(args) - 1>{});
}
